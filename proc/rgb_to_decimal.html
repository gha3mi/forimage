   elemental pure subroutine rgb_to_decimal(r, g, b, decimal)
      implicit none
      integer(ik), intent(in)  :: r, g, b
      integer(ik), intent(out) :: decimal

      decimal = r*65536_ik + g*256_ik + b
   end subroutine rgb_to_decimal
